\cs_new_protected:Nn \__comm_record:nnnn {
  \tl_set:Nn \l_comm_side_tl { R } % default
  \cs_if_exist:cT { commside #1 } {
    \tl_set:Nx \l_comm_side_tl { \use:c { commside #1 } }
  }
  
  \seq_if_exist:cF { g__comm_page_ids_seq_ #2 _ \l_comm_side_tl } {
    \seq_new:c { g__comm_page_ids_seq_ #2 _ \l_comm_side_tl }
    \seq_gput_right:Nx \g__comm_pages_seq { #2 _ \l_comm_side_tl }
  }
  \seq_gput_right:cn { g__comm_page_ids_seq_ #2 _ \l_comm_side_tl } { #1 }
  \prop_gput:Nnn \g__comm_src_y_prop { #1 } { #3 }
  \prop_gput:Nnn \g__comm_box_ht_prop { #1 } { #4 }
}

\cs_new_protected:Nn \__comm_solve_elasticity: {
  \typeout{Starting~solve~elasticity}
  \dim_set:Nn \l_comm_top_margin_dim { \paperheight - 0.2in }
  \dim_set:Nn \l_comm_bottom_margin_dim { 0.2in }
  \typeout{Margins~set:~top=\the\l_comm_top_margin_dim,~bottom=\the\l_comm_bottom_margin_dim}
  
  \seq_map_inline:Nn \g__comm_pages_seq {
    \typeout{Processing~page-side~##1}
    \group_begin:
    \bool_set_true:N \l_comm_valid_coords_bool
    
    % Check if all comments on this page-side have valid (non-zero) coordinates
    \seq_map_inline:cn { g__comm_page_ids_seq_ ##1 } {
      \dim_set:Nn \l_y_dim { \prop_item:Nn \g__comm_src_y_prop { ####1 } }
      \dim_compare:nNnT { \l_y_dim } = { 0pt } {
        \bool_set_false:N \l_comm_valid_coords_bool
      }
    }
    
    \bool_if:NT \l_comm_valid_coords_bool {
      \prop_set_eq:NN \l_curr_y_prop \g__comm_src_y_prop
      
      % Relaxation Loop
      \int_step_inline:nnnn { 1 } { 1 } { 30 } {
        % Boundary constraints (only if page-side has > 1 comment)
        \int_compare:nT { \seq_count:c { g__comm_page_ids_seq_ ##1 } > 1 } {
          % First comment (highest)
          \tl_set:Nx \l_id_first { \seq_item:cn { g__comm_page_ids_seq_ ##1 } { 1 } }
          \dim_set:Nn \l_y_dim { \prop_item:No \l_curr_y_prop { \l_id_first } }
          \dim_set:Nn \l_ht_dim { \prop_item:No \g__comm_box_ht_prop { \l_id_first } }
          \dim_compare:nNnT { \l_y_dim + \l_ht_dim / 2 } > { \l_comm_top_margin_dim } {
            \typeout{Pushing~highest~comment~\l_id_first~down}
            \prop_put:Nox \l_curr_y_prop { \l_id_first } { \dim_eval:n { \l_comm_top_margin_dim - \l_ht_dim / 2 } }
          }
          
          % Last comment (lowest)
          \tl_set:Nx \l_id_last { \seq_item:cn { g__comm_page_ids_seq_ ##1 } { \seq_count:c { g__comm_page_ids_seq_ ##1 } } }
          \dim_set:Nn \l_y_dim { \prop_item:No \l_curr_y_prop { \l_id_last } }
          \dim_set:Nn \l_ht_dim { \prop_item:No \g__comm_box_ht_prop { \l_id_last } }
          \dim_compare:nNnT { \l_y_dim - \l_ht_dim / 2 } < { \l_comm_bottom_margin_dim } {
            \typeout{Pushing~lowest~comment~\l_id_last~up}
            \prop_put:Nox \l_curr_y_prop { \l_id_last } { \dim_eval:n { \l_comm_bottom_margin_dim + \l_ht_dim / 2 } }
          }
        }
        
        % Pairwise repulsion (only if page-side has > 1 comment)
        \int_compare:nT { \seq_count:c { g__comm_page_ids_seq_ ##1 } > 1 } {
          \int_step_inline:nnnn { 1 } { 1 } { \seq_count:c { g__comm_page_ids_seq_ ##1 } - 1 } {
            \tl_set:Nx \l_id_a { \seq_item:cn { g__comm_page_ids_seq_ ##1 } { ########1 } }
            \tl_set:Nx \l_id_b { \seq_item:cn { g__comm_page_ids_seq_ ##1 } { \int_eval:n { ########1 + 1 } } }
            
            \dim_set:Nn \l_y_a { \prop_item:No \l_curr_y_prop { \l_id_a } }
            \dim_set:Nn \l_y_b { \prop_item:No \l_curr_y_prop { \l_id_b } }
            
            \dim_set:Nn \l_ht_a { \prop_item:No \g__comm_box_ht_prop { \l_id_a } }
            \dim_set:Nn \l_ht_b { \prop_item:No \g__comm_box_ht_prop { \l_id_b } }
            
            \dim_set:Nn \l_min_dist { \l_ht_a / 2 + \l_ht_b / 2 + 3pt }
            \dim_set:Nn \l_curr_dist { \l_y_a - \l_y_b }
            
            \dim_compare:nNnT { \l_curr_dist } < { \l_min_dist } {
              \dim_set:Nn \l_overlap { \l_min_dist - \l_curr_dist }
              \prop_put:Nox \l_curr_y_prop { \l_id_a } { \dim_eval:n { \l_y_a + \l_overlap / 2 } }
              \prop_put:Nox \l_curr_y_prop { \l_id_b } { \dim_eval:n { \l_y_b - \l_overlap / 2 } }
            }
          }
        }
      }
      
      % Save final calculated shifts
      \seq_map_inline:cn { g__comm_page_ids_seq_ ##1 } {
        \dim_set:Nn \l_y_orig { \prop_item:Nn \g__comm_src_y_prop { ####1 } }
        \dim_set:Nn \l_y_final { \prop_item:Nn \l_curr_y_prop { ####1 } }
        \dim_set:Nn \l_shift { \l_y_orig - \l_y_final }
        \tl_gset:cx { g__comm_shift_ ####1 _dim } { \dim_use:N \l_shift }
      }
    }
    \group_end:
  }
}
